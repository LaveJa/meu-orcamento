<div class="card">
    <h2>Calculadora</h2>
    
    <label>Valor de Custo (R$)</label>
    <input type="number" id="custo" placeholder="0,00" oninput="calcular()">
    
    <label>Margem de Lucro (%)</label>
    <input type="number" id="margem" placeholder="Ex: 30" oninput="calcular()">

    <div class="result-area">
        <small>Preço de Venda</small>
        <span id="total">R$ 0,00</span>
    </div>
</div>

<script>
    function calcular() {
        const custo = parseFloat(document.getElementById('custo').value) || 0;
        const margem = parseFloat(document.getElementById('margem').value) || 0;
        
        // Exemplo de fórmula: Custo + Porcentagem
        const resultado = custo + (custo * (margem / 100));
        
        document.getElementById('total').innerText = resultado.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' });
    }

    // Registra o Service Worker para funcionar offline
    if ('serviceWorker' in navigator) {
        navigator.serviceWorker.register('sw.js');
    }
</script>
